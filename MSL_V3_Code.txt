volatile uint16_t pulseWidth = 1051;
volatile uint32_t lastRise = 0;


ISR(PCINT0_vect) {
  if (PINB & (1 << PB2)) {   // rising edge on pin 2 (PB2)
    lastRise = micros();
  } else {                   // falling edge
    pulseWidth = micros() - lastRise;
  }
}

void loop()
{
  uint16_t val = pulseWidth; // copy safely
  //For t12 Tx = (min = 1051pwm; mid = 1501pwm; max = 1951pwm)
  //Read Accuracy = 150pwm; 
  //lower limt = min + Read Accuracy => 1051+150 = 1201
  //Upper limt = mid - Read Accuracy => 1501-150 = 1351
  //Safe Control set at = (lower limt + Upperlimt)/2 => (1201+1351)/2 = 1276
  
  if (val >= 800 && val <= 1276) // Master Active
  { 
        digitalWrite(3, LOW);

        leds_grn[0] = CRGB(0,255,0); //green
        leds_grn[1] = CRGB(0,255,0);
        leds_grn[2] = CRGB(0,255,0);
        leds_grn[3] = CRGB(0,255,0);

        leds_red[0] = CRGB(255,0,0); //red
        leds_red[1] = CRGB(255,0,0);
        leds_red[2] = CRGB(255,0,0);
        leds_red[3] = CRGB(255,0,0);
           
        
  } 
  else if(val > 1276 && val <= 2200) // Slave Active
  {
        digitalWrite(3, HIGH);

        leds_grn[0] = CRGB(0,0,255); //blue
        leds_grn[1] = CRGB(0,0,255);
        leds_grn[2] = CRGB(0,0,255);
        leds_grn[3] = CRGB(0,0,255);

        leds_red[0] = CRGB(255,0,0); //red
        leds_red[1] = CRGB(255,0,0);
        leds_red[2] = CRGB(255,0,0);
        leds_red[3] = CRGB(255,0,0);
        
         
  }
  FastLED.show();
}
